{{- $assetsDir := getenv "HUGO_ASSETS_DIR" | default "_assets" -}} {{- $destination := .Destination -}} {{- $url := urls.Parse $destination -}} {{- $assetsPathPrefix := printf "%s/" $assetsDir -}} {{- if not $url.Host -}} {{- $path := $url.Path -}} {{- if strings.HasPrefix $path (printf "./%s" $assetsPathPrefix) -}} {{- $path = strings.TrimPrefix "./" $path -}} {{- end -}} {{- if strings.HasPrefix $path $assetsPathPrefix -}} {{- $destination = printf "/%s" $path -}} {{- with $url.RawQuery -}} {{- $destination = printf "%s?%s" $destination . -}} {{- end -}} {{- with $url.Fragment -}} {{- $destination = printf "%s#%s" $destination . -}} {{- end -}} {{- end -}} {{- end -}} {{- $alt := .Text | plainify -}} {{- if .Title -}} {{- else -}} {{- end -}}
